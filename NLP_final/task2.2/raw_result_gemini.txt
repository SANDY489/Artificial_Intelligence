import numpy as np

def relation_detection(sentence1, sentence2):
  """
  Determine the relation between two sentences.

  Args:
    sentence1: The first sentence.
    sentence2: The second sentence.

  Returns:
    0 if there is no relation between the sentences, 1 if sentence1 supports
    sentence2, and 2 if sentence1 attacks sentence2.
  """

  # Convert the sentences to lowercase and tokenize them.
  sentence1 = sentence1.lower().split()
  sentence2 = sentence2.lower().split()

  # Create a list of stop words.
  stop_words = ["a", "an", "the", "is", "are", "was", "were", "be", "been",
                 "being", "have", "has", "had", "having", "do", "does", "did",
                 "doing", "would", "should", "could", "may", "might", "must",
                 "can", "cannot", "will", "shall", "should", "ought", "and",
                 "but", "or", "nor", "for", "against", "on", "off", "in", "out",
                 "over", "under", "above", "below", "to", "from", "by", "with",
                 "as", "like", "because", "since", "after", "before", "when",
                 "where", "how", "what", "who", "whom", "whose", "that",
                 "which", "this", "that", "these", "those", "here", "there",
                 "everywhere", "anywhere", "nowhere", "some", "any", "all",
                 "every", "no", "none", "only", "both", "either", "neither",
                 "few", "many", "most", "less", "more", "much", "little",
                 "up", "down", "left", "right", "forward", "backward", "away",
                 "towards"]

  # Remove the stop words from the sentences.
  sentence1 = [word for word in sentence1 if word not in stop_words]
  sentence2 = [word for word in sentence2 if word not in stop_words]

  # Create a list of the words that appear in both sentences.
  common_words = set(sentence1) & set(sentence2)

  # If there are no common words, then there is no relation between the sentences.
  if len(common_words) == 0:
    return 0

  # If the first sentence contains more common words than the second sentence,
  # then the first sentence supports the second sentence.
  elif len(common_words) > len(sentence2):
    return 1

  # Otherwise, the first sentence attacks the second sentence.
  else:
    return 2

import numpy as np

def relation_detection(sent1, sent2):
  """
  This function takes two sentences as input and returns a numeric value indicating the relationship between the two sentences.

  Args:
    sent1 (str): The first sentence.
    sent2 (str): The second sentence.

  Returns:
    int: A numeric value indicating the relationship between the two sentences.
  """
  # Convert the sentences to lowercase and tokenize them.
  sent1_tokens = sent1.lower().split()
  sent2_tokens = sent2.lower().split()

  # Create a matrix of the cosine similarities between the tokens in the two sentences.
  cosine_similarities = np.zeros((len(sent1_tokens), len(sent2_tokens)))
  for i in range(len(sent1_tokens)):
    for j in range(len(sent2_tokens)):
      cosine_similarities[i, j] = cosine_similarity(sent1_tokens[i], sent2_tokens[j])

  # Find the maximum cosine similarity between any two tokens in the two sentences.
  max_cosine_similarity = np.max(cosine_similarities)

  # If the maximum cosine similarity is greater than 0.5, then the two sentences are considered to be related.
  if max_cosine_similarity > 0.5:
    # Determine the type of relationship between the two sentences.
    if sent1_tokens[-1] == '.' and sent2_tokens[0].isupper():
      # The first sentence is likely a supporting statement for the second sentence.
      return 1
    elif sent1_tokens[-1] == '?' and sent2_tokens[0].islower():
      # The first sentence is likely an attacking statement against the second sentence.
      return 2

  # If the maximum cosine similarity is not greater than 0.5, then the two sentences are not considered to be related.
  return 0

import numpy as np

def relation_detection(sentence1, sentence2):
  """
  This function takes two sentences as input and returns a numeric value indicating the relationship between the two sentences.

  Args:
    sentence1: The first sentence.
    sentence2: The second sentence.

  Returns:
    A numeric value indicating the relationship between the two sentences. 0 indicates no relation, 1 indicates a Supporting relation, and 2 indicates an Attacking relation.
  """

  # Convert the sentences to lowercase and tokenize them.
  sentence1 = sentence1.lower().split()
  sentence2 = sentence2.lower().split()

  # Create a list of stopwords.
  stopwords = set(np.loadtxt("stopwords.txt", dtype=str))

  # Remove stopwords from the sentences.
  sentence1 = [word for word in sentence1 if word not in stopwords]
  sentence2 = [word for word in sentence2 if word not in stopwords]

  # Create a list of keywords for each sentence.
  keywords1 = [word for word in sentence1 if word in ["good", "great", "excellent", "positive", "favorable"]]
  keywords2 = [word for word in sentence2 if word in ["bad", "terrible", "awful", "negative", "unfavorable"]]

  # Check if the two sentences have any keywords in common.
  if len(set(keywords1).intersection(set(keywords2))) > 0:
    # If the two sentences have any keywords in common, then they are related.
    return 1
  else:
    # If the two sentences do not have any keywords in common, then they are not related.
    return 0
